Issue 1 — Trip Timeout Stored in Memory CRITICAL

Problem

In tripMatchingService.js, the expiry timeout for each trip is stored inside a JavaScript Map in the process's heap:

this.activeTimeouts = new Map();  // inside TripMatchingService constructor

const timeoutId = setTimeout(async () => {
    await this._checkTripTimeout(tripId, io);
    this.activeTimeouts.delete(tripId);
}, this.offerTtlMs);

If the server crashes, restarts, or deploys while trips are in SEARCHING state, all stored timeouts vanish. The passenger never receives the trip:no_drivers event and is permanently stuck on the loading screen. The Redis key trip:timeout:{tripId} exists but is never read back to restore the timeout — it only acts as a "was this already accepted?" marker.

Fix — Replace in-memory setTimeout with a Redis-backed expiry check

Step 1: Install BullMQ (recommended) or use Redis keyspace notifications.

npm install bullmq

Step 2: Create src/queues/tripTimeout.queue.js

const { Queue, Worker } = require('bullmq');
const { redisClient }   = require('../config/redis');

const QUEUE_NAME = 'trip-timeout';

const tripTimeoutQueue = new Queue(QUEUE_NAME, {
    connection: { host: process.env.REDIS_HOST, port: process.env.REDIS_PORT || 6379 },
});

// Call this instead of setTimeout in tripMatchingService.js
async function scheduleTripTimeout(tripId, delayMs) {
    await tripTimeoutQueue.add('expire', { tripId }, {
        delay:    delayMs,
        jobId:    `timeout:${tripId}`,   // idempotent — same tripId = same job
        removeOnComplete: true,
        removeOnFail:     true,
    });
}

// Call this when a driver accepts to cancel the scheduled timeout
async function cancelTripTimeout(tripId) {
    const job = await tripTimeoutQueue.getJob(`timeout:${tripId}`);
    if (job) await job.remove();
}

module.exports = { tripTimeoutQueue, scheduleTripTimeout, cancelTripTimeout };

Step 3: Create src/queues/tripTimeout.worker.js — start this in server.js

const { Worker }         = require('bullmq');
const { redisClient, redisHelpers, REDIS_KEYS } = require('../config/redis');

function startTripTimeoutWorker(io) {
    const worker = new Worker('trip-timeout', async (job) => {
        const { tripId } = job.data;
        const tripData   = await redisHelpers.getJson(REDIS_KEYS.ACTIVE_TRIP(tripId));
        if (!tripData || tripData.status !== 'SEARCHING') return; // already accepted

        // Clean up Redis
        await redisClient.del(REDIS_KEYS.ACTIVE_TRIP(tripId));
        await redisClient.del(`passenger:active_trip:${tripData.passengerId}`);
        await redisClient.del(REDIS_KEYS.TRIP_OFFERS(tripId));

        // Notify passenger
        io.to(`passenger:${tripData.passengerId}`).emit('trip:no_drivers', {
            tripId,
            message:   'No drivers accepted your trip. Please try again.',
            timestamp: new Date().toISOString(),
        });
    }, {
        connection: { host: process.env.REDIS_HOST, port: process.env.REDIS_PORT || 6379 },
    });

    console.log('✅ [WORKER] Trip timeout worker started');
    return worker;
}

module.exports = { startTripTimeoutWorker };

Step 4: In tripMatchingService.js, replace the setTimeout block:

// REMOVE THIS:
const timeoutId = setTimeout(async () => { ... }, this.offerTtlMs);
this.activeTimeouts.set(tripId, timeoutId);

// REPLACE WITH:
const { scheduleTripTimeout } = require('../queues/tripTimeout.queue');
await scheduleTripTimeout(tripId, this.offerTtlMs);

Step 5: In acceptTrip(), replace the clearTimeout block:

// REMOVE THIS:
const timeoutId = this.activeTimeouts.get(tripId);
if (timeoutId) { clearTimeout(timeoutId); this.activeTimeouts.delete(tripId); }

// REPLACE WITH:
const { cancelTripTimeout } = require('../queues/tripTimeout.queue');
await cancelTripTimeout(tripId);

Step 6: In server.js, start the worker after Socket.IO is ready:

const { startTripTimeoutWorker } = require('./src/queues/tripTimeout.worker');
startTripTimeoutWorker(io);

Issue 2 — No Socket.IO Redis Adapter CRITICAL

Problem

Socket.IO uses its default in-memory adapter. Events emitted on one process (e.g., io.to('driver:X').emit()) are invisible to drivers connected to any other process. You cannot run more than one Node.js instance. There is no fault tolerance — a server crash disconnects every user simultaneously. You also cannot do zero-downtime deployments.

Fix — Add @socket.io/redis-adapter

Step 1: Install the adapter.

npm install @socket.io/redis-adapter

Step 2: In src/sockets/index.js, add the adapter immediately after creating the Server:

const { Server }        = require('socket.io');
const { createAdapter } = require('@socket.io/redis-adapter');
const { createClient }  = require('ioredis');   // already a dependency

function setupSocketIO(server) {
    const io = new Server(server, {
        cors: { origin: process.env.CORS_ORIGIN || '*', methods: ['GET', 'POST'] },
        pingTimeout:    60000,
        pingInterval:   25000,
        connectTimeout: 45000,
    });

    // ── ADD THIS BLOCK ───────────────────────────────────────────────────
    const pubClient = new createClient(process.env.REDIS_URL || 'redis://localhost:6379');
    const subClient = pubClient.duplicate();

    Promise.all([pubClient.connect(), subClient.connect()]).then(() => {
        io.adapter(createAdapter(pubClient, subClient));
        console.log('✅ [SOCKET] Redis adapter attached — horizontal scaling enabled');
    });
    // ────────────────────────────────────────────────────────────────────

    // ... rest of existing setup unchanged
}

What this unlocks: You can now run multiple instances behind an Nginx / load balancer and events broadcast from any instance reach every connected client regardless of which instance they're connected to.

Issue 3 — Delivery Redis Key Mismatch — Direct Socket Lookup Always Fails CRITICAL

Problem

In src/sockets/delivery/deliverySocket.service.js, the fallback socket lookup uses:

const socketId = await redisClient.get(`socket:user:${senderUuid}`);

But throughout the rest of the codebase (including src/config/redis.js where REDIS_KEYS.USER_SOCKET is defined), socket IDs are stored under the key:

user:socket:${userId}   // e.g.  user:socket:abc-123

The key pattern is inverted. The delivery service looks for socket:user:X but the value is stored at user:socket:X. The lookup returns null every time, silently. The sender only receives delivery events via room broadcasts, and if they are not in the room, they receive nothing.

Fix — Use REDIS_KEYS.USER_SOCKET consistently

In src/sockets/delivery/deliverySocket.service.js, change the emitToSender helper:

// BEFORE:
const socketId = await redisClient.get(`socket:user:${senderUuid}`);

// AFTER:
const { REDIS_KEYS } = require('../../config/redis');
const socketId = await redisClient.get(REDIS_KEYS.USER_SOCKET(senderUuid));

Then search the entire codebase for any other use of the socket:user: pattern and replace them all with REDIS_KEYS.USER_SOCKET(id).

Issue 4 — Triple-Emit Sends Duplicate Events HIGH

Problem

In both broadcastTripToDrivers() and acceptTrip(), the same event is emitted to three targets for the same user:

io.to(`driver:${driverId}`).emit('trip:new_request', offer);  // room 1
io.to(`user:${driverId}`).emit('trip:new_request', offer);    // room 2
io.to(socketId).emit('trip:new_request', offer);              // direct socket

On connection, every user joins both driver:{id} (or passenger:{id}) AND user:{id}. The direct socket is also in both of those rooms. This means a single event delivery results in 2–3 duplicate socket messages arriving at the client. The mobile app must handle and discard duplicates, which it likely does not, causing the trip offer screen to appear multiple times or accept to be called twice.

Fix — Emit to one room only

Choose one canonical room per user type and emit only to that. Remove the other two channels.

// For drivers — replace all 3 lines with:
io.to(`driver:${driverId}`).emit('trip:new_request', offerWithDistance);

// For passengers — replace all 3 lines with:
io.to(`passenger:${tripData.passengerId}`).emit('trip:driver_assigned', assignmentData);

The room-based emit already reaches every socket that joined that room — no need for a redundant direct socket emit on top of it. Apply this change consistently throughout tripMatchingService.js and driverHandlers.js for all events: trip:new_request, trip:driver_assigned, trip:request_expired, trip:no_drivers, and trip:state_sync.

Note: The only case where a direct socket emit is justified is if the user failed to join their room (e.g., room join threw an error). Since that is handled in the connection callback with a try/catch and log, you can safely drop the redundant direct socket emit path entirely.

Issue 5 — MySQL Write on Every GPS Ping HIGH

Problem

In src/sockets/driverHandlers.js, handleDriverLocationUpdate fires a MySQL UPDATE on every GPS event:

DriverLocation.update(
    { lat, lng, heading, speed, accuracy, last_updated: new Date() },
    { where: { driver_id: socket.userId } }
).catch(err => console.error(...));

This is fire-and-forget so it does not block the socket handler, but it still executes. With 200 active drivers sending a location update every 3–5 seconds, this generates 40–60 MySQL writes per second on a single row per driver. At 10k users with more drivers, this number rises proportionally. The DriverLocation table is not used for real-time decisions — only for the admin live monitor and historical analytics — so sub-second freshness is not required.

Fix — Debounce the MySQL write to every 15 seconds per driver

Add a per-driver debounce using a Redis timestamp. Write to MySQL only when at least 15 seconds have passed since the last write for that driver.

// In handleDriverLocationUpdate, replace the DriverLocation.update() block with:

const MYSQL_FLUSH_INTERVAL_S = 15;
const lastFlushKey = `driver:location:last_mysql_flush:${socket.userId}`;

const lastFlush = await redisClient.get(lastFlushKey);
const now       = Date.now();

if (!lastFlush || (now - parseInt(lastFlush)) >= MYSQL_FLUSH_INTERVAL_S * 1000) {
    // Time to flush to MySQL
    DriverLocation.update(
        { lat, lng, heading, speed, accuracy, last_updated: new Date() },
        { where: { driver_id: socket.userId } }
    ).catch(err => console.error('⚠️ [LOCATION] MySQL flush error:', err.message));

    await redisClient.setex(lastFlushKey, MYSQL_FLUSH_INTERVAL_S + 5, now.toString());
}

This reduces MySQL location writes from ~50/second to ~13/second at 200 drivers — a 75% reduction — with no change in real-time behavior since Redis is always updated on every ping.

Issue 6 — _getPassengerRating Loads Every Row Into Memory HIGH

Problem

In tripMatchingService.js:

const rows = await Rating.findAll({
    where:      { ratedUser: passengerId, ratingType: 'DRIVER_TO_PASSENGER' },
    attributes: ['rating'],
});
const avg = rows.reduce((s, r) => s + r.rating, 0) / rows.length;

This fetches every rating row for the passenger into a JavaScript array, then computes the average in Node.js. A passenger with 500 completed trips will transfer 500 rows from MySQL to the server for a calculation that MySQL can do in a single query with zero data transfer overhead. This also runs on every trip broadcast.

Fix — Use a SQL AVG aggregate query

const { fn, col, literal } = require('sequelize');

async _getPassengerRating(passengerId) {
    try {
        const result = await Rating.findOne({
            where: {
                ratedUser:  passengerId,
                ratingType: 'DRIVER_TO_PASSENGER',
            },
            attributes: [
                [fn('AVG', col('rating')), 'avg_rating'],
                [fn('COUNT', col('id')),   'count'],
            ],
            raw: true,
        });

        if (!result || !result.count || parseInt(result.count) === 0) return null;
        return parseFloat(parseFloat(result.avg_rating).toFixed(1));

    } catch {
        return null;
    }
}

One SQL query, no data transferred beyond two numbers, runs in under 1ms with an index on (ratedUser, ratingType).

Issue 7 — Driver Decline Is Not Tracked MEDIUM

Problem

In driverHandlers.js, handleTripDecline only acknowledges the event:

async function handleTripDecline(socket, data) {
    socket.emit('trip:decline:success', { tripId, message: 'Trip declined' });
    // Nothing else
}

The declined driver is never tracked. On reconnect, _replayMissedDriverEvents checks driver:pending_offers:{driverId} and re-sends the offer. A driver who explicitly declined will receive the same trip again after their next reconnect. There is also no mechanism to exclude a decliner from the next re-broadcast cycle.

Fix — Record the decline in Redis and filter declined drivers

Step 1: In handleTripDecline, record the driver as having declined:

async function handleTripDecline(socket, data) {
    const { tripId, reason = 'Driver declined' } = data || {};
    const driverId = socket.userId;

    if (tripId) {
        // Add to the trip's declined set (expires with the trip offer TTL)
        await redisClient.sadd(REDIS_KEYS.TRIP_DECLINED(tripId), driverId);
        await redisClient.expire(REDIS_KEYS.TRIP_DECLINED(tripId), 300); // 5 min TTL

        // Remove from this driver's pending offers so replay doesn't re-send it
        const pending = await redisHelpers.getJson(`driver:pending_offers:${driverId}`) || [];
        const updated = pending.filter(o => o.tripId !== tripId);
        await redisHelpers.setJson(`driver:pending_offers:${driverId}`, updated, 300);

        console.log(`🚫 [SOCKET-DRIVER] Driver ${driverId} declined trip ${tripId}`);
    }

    socket.emit('trip:decline:success', { tripId, message: 'Trip declined' });
}

Step 2: In broadcastTripToDrivers(), add REDIS_KEYS.TRIP_DECLINED to the filters:

// After filtering by ride mode and wallet balance, add:
const declinedSet    = await redisClient.smembers(REDIS_KEYS.TRIP_DECLINED(tripId));
const declinedIds    = new Set(declinedSet);
const finalDrivers   = eligibleDrivers.filter(d => !declinedIds.has(d.driverId));

console.log(`✅ [MATCHING] ${finalDrivers.length} drivers after excluding decliners`);

Issue 8 — No Re-Broadcast With Expanding Radius MEDIUM

Problem

When broadcastTripToDrivers() finds zero eligible drivers, it immediately deletes the trip from Redis and emits trip:no_drivers. The passenger has to manually retry from scratch. In a city with sparse driver coverage at certain times, this produces a very poor experience — a trip fails before any driver even has a chance to move closer.

Fix — Retry with an expanding search radius

Instead of failing immediately, schedule a re-broadcast at a larger radius. Use the BullMQ queue from Fix #1 for this too.

In tripMatchingService.js, modify the "no drivers" path:

// Current failure path (REMOVE the del + return):
if (rideModeDrivers.length === 0) {
    await redisClient.del(REDIS_KEYS.ACTIVE_TRIP(tripId));
    await redisClient.del(`passenger:active_trip:${trip.passengerId}`);
    return { success: false, reason: 'No drivers available' };
}

// REPLACE WITH — read the current attempt number from the trip data:
if (eligibleDrivers.length === 0) {
    const attempt = (trip.broadcastAttempt || 0) + 1;
    const RADIUS_BY_ATTEMPT = [5, 8, 12]; // km per attempt

    if (attempt > RADIUS_BY_ATTEMPT.length) {
        // All attempts exhausted — give up
        await redisClient.del(REDIS_KEYS.ACTIVE_TRIP(tripId));
        await redisClient.del(`passenger:active_trip:${trip.passengerId}`);
        io.to(`passenger:${trip.passengerId}`).emit('trip:no_drivers', {
            tripId,
            message: 'No drivers available right now. Please try again later.',
            timestamp: new Date().toISOString(),
        });
        return { success: false, reason: 'No drivers available after all retries' };
    }

    // Update trip with new radius attempt and reschedule
    trip.broadcastAttempt = attempt;
    trip.searchRadiusKm   = RADIUS_BY_ATTEMPT[attempt - 1];
    await redisHelpers.setJson(REDIS_KEYS.ACTIVE_TRIP(tripId), trip, 300);

    // Notify passenger that we are still searching
    io.to(`passenger:${trip.passengerId}`).emit('trip:still_searching', {
        tripId,
        attempt,
        message:    'Still searching for a driver nearby...',
        searchingKm: trip.searchRadiusKm,
    });

    // Schedule a re-broadcast in 15 seconds
    await scheduleTripTimeout(tripId + ':retry', 15000);  // or a separate retry queue

    return { success: false, reason: 'No drivers in current radius — retrying', retrying: true };
}

Issue 9 — deliverySocket.service.js Not Connected to the Location Pipeline HIGH

Problem

The file src/sockets/delivery/deliverySocket.service.js contains a clear comment:

// Usage in locationHandlers.js:
//   if (socket.currentMode === 'delivery') {

const deliverySocketService = require('./delivery/deliverySocket.service');

async function handleDriverLocationUpdate(socket, data, io) {

        if (!lat || !lng || lat < -90 || lat > 90 || lng < -180 || lng > 180) return;

        // 1. Update Redis geo index (always — for both ride and delivery)
        await _updateDriverLocationInRedis(socket.userId, lat, lng, heading, speed, accuracy);

        // 2. Debounced MySQL flush (see Fix #5)
        // ... debounce logic here ...

        // 3. RIDE MODE: broadcast to passenger if there is an active trip
        if (socket.activeMode === 'DRIVER') {
            const activeTrip = await Trip.findOne({
                where: {

    } catch (error) {
        console.error('❌ [SOCKET-DRIVER] Location update error:', error.message);
    }
}

Issue 10 — No Distributed Lock Visible for Delivery Acceptance MEDIUM

Problem

The ride-hailing flow uses a Redis NX lock in acceptTrip() to prevent two drivers from accepting the same trip simultaneously. The delivery acceptance flow (in src/controllers/delivery.controller.js) does not appear to have an equivalent guard. If two delivery agents tap "Accept" at the same moment for the same delivery, a race condition could assign the delivery to both, or produce a corrupted state where the first agent to write wins but the second agent's UI shows them as accepted.

Fix — Add a Redis NX lock to delivery acceptance

In src/controllers/delivery.controller.js (or wherever delivery acceptance is handled), wrap the acceptance logic in the same lock pattern used for trips:

const { v4: uuidv4 }    = require('uuid');
const { redisClient }   = require('../config/redis');

async function acceptDelivery(req, res) {
    const { deliveryId } = req.params;
    const agentId        = req.user.uuid;

    const lockKey   = `delivery:lock:${deliveryId}`;
    const lockValue = uuidv4();

    // Acquire lock (10 second TTL — auto-expires if we crash mid-flight)
    const lockAcquired = await redisClient.set(lockKey, lockValue, 'EX', 10, 'NX');

    if (!lockAcquired) {
        return res.status(409).json({
            success: false,
            message: 'This delivery is already being accepted by another agent.',
            code:    'DELIVERY_LOCK_CONTENTION',
        });
    }

    try {
        // ── Verify delivery is still available ─────────────────────────
        const delivery = await Delivery.findByPk(deliveryId);

        if (!delivery || delivery.status !== 'pending') {
            return res.status(409).json({
                success: false,
                message: 'Delivery is no longer available.',
                code:    'DELIVERY_NOT_AVAILABLE',
            });
        }

        // ── Assign to agent ────────────────────────────────────────────
        delivery.driver_id   = agentId;
        delivery.status      = 'accepted';
        delivery.accepted_at = new Date();
        await delivery.save();

        // ── Cache in Redis ─────────────────────────────────────────────
        await redisClient.setex(
            `driver:active_delivery:${agentId}`,
            7200,
            String(deliveryId)
        );

        // ... notify sender, emit socket events ...

        return res.json({ success: true, delivery });

    } finally {
        // Release lock — only if we still own it
        const current = await redisClient.get(lockKey);
        if (current === lockValue) await redisClient.del(lockKey);
    }
}
